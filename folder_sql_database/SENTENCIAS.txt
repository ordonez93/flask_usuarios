*******************Punto 1. (USUARIO ENCUESTA CANT_SECCIONES CANT_PREGUNTAS)

SELECT concat_ws(' ',usuarios.nombres, usuarios.apellidos) as 'usuario', cantidad_encuestas.nombre as 'encuesta', cantidad_secciones.cantidad as 'secciones', cantidad_preguntas.cantidad as 'preguntas' FROM usuarios
INNER JOIN (
    SELECT usuarios.id,COUNT(encuestas.id) as "cantidad",encuestas.nombre as "nombre"from usuarios
    LEFT JOIN encuestas on encuestas.usuario_id = usuarios.id
    GROUP BY usuarios.id
    ORDER BY cantidad
) as cantidad_encuestas
ON 
cantidad_encuestas.id = usuarios.id
INNER JOIN (
    SELECT usuarios.id,COUNT(secciones.id) as "cantidad" from usuarios
    LEFT JOIN encuestas on encuestas.usuario_id = usuarios.id
    LEFT JOIN secciones on  secciones.encuesta_id = encuestas.id
    GROUP BY usuarios.id
    ORDER BY cantidad
) as cantidad_secciones
ON
cantidad_secciones.id = usuarios.id
INNER JOIN(
    SELECT usuarios.id,COUNT(preguntas.id) as "cantidad" FROM usuarios
    LEFT JOIN encuestas on encuestas.usuario_id = usuarios.id
    LEFT JOIN secciones on  secciones.encuesta_id = encuestas.id
    LEFT JOIN preguntas ON preguntas.seccion_id = secciones.id
    GROUP BY usuarios.id
    ORDER BY cantidad
) as cantidad_preguntas
ON 
cantidad_preguntas.id = usuarios.id;
